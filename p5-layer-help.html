üöÄ Getting Started

1. Adding a P5 Layer

  1. Open the Layer Panel in the left sidebar
  2. Click "Add P5 Layer"
  3. The default sketch will start running immediately
  4. Click "P5 Code" in the transport bar to edit the sketch

2. Basic Sketch Structure

function setup() {
    createCanvas(windowWidth, windowHeight);
    // Your initialization code here
}

function draw() {
    // Expose parameters for MIDI control
    let speed = p5Param('speed', 1.0, { min: 0.1, max: 5.0, label: 'Animation Speed' });
    let size = p5Param('size', 100, { min: 20, max: 300, label: 'Circle Size' });
    
    background(0);
    
    // Use the parameters in your sketch
    let x = width/2 + cos(frameCount * 0.02 * speed) * 150;
    let y = height/2 + sin(frameCount * 0.02 * speed) * 150;
    circle(x, y, size);
}

function windowResized() {
    resizeCanvas(windowWidth, windowHeight);
}

‚öôÔ∏è Parameter System

The p5Param() Function

Key Concept: Use let (not const) when storing parameter values, or call p5Param() directly in your code.

Syntax:

let value = p5Param(name, defaultValue, options);
Parameter Type Description Required
name String Unique identifier for the parameter ‚úÖ Yes
defaultValue Number Initial value when no MIDI input ‚úÖ Yes
options.min Number Minimum value for the parameter ‚ùå No
options.max Number Maximum value for the parameter ‚ùå No
options.step Number Step size for discrete values ‚ùå No
options.label String Human-readable name for UI display ‚ùå No

üéõÔ∏è Parameter Types

Continuous Values

// Speed control (0.1 to 5.0)
let speed = p5Param('speed', 1.0, { 
    min: 0.1, 
    max: 5.0, 
    label: 'Animation Speed' 
});

// Size with large range
let size = p5Param('objectSize', 50, { 
    min: 10, 
    max: 500, 
    label: 'Object Size' 
});

// Color hue (0-360 degrees)
let hue = p5Param('hue', 180, { 
    min: 0, 
    max: 360, 
    label: 'Color Hue' 
});

Discrete Values

// Number of objects (integer steps)
let count = p5Param('count', 10, { 
    min: 1, 
    max: 50, 
    step: 1,
    label: 'Object Count' 
});

// Use Math.floor() to ensure integer
for (let i = 0; i < Math.floor(count); i++) {
    // Draw objects
}

Boolean-like Values

// Toggle behavior (0-1 range, treat as boolean)
let showTrails = p5Param('trails', 0, { 
    min: 0, 
    max: 1, 
    step: 1,
    label: 'Show Trails' 
});

if (showTrails > 0.5) {
    // Semi-transparent background for trails
    background(0, 20);
} else {
    // Solid background
    background(0);
}

üéπ MIDI Mapping

How to Map Parameters to MIDI

  1. Expose Parameters: Use p5Param() in your sketch
  2. Add MIDI Control: Click "Add CC Control" in MIDI drawer
  3. Select P5 Parameter: Choose from the P5 dropdown
  4. Learn MIDI: Click "Learn" and move your controller
  5. Test: Move the controller to see real-time changes
‚úÖ Pro Tip: You can map the same MIDI controller to both a grid parameter AND a P5 parameter simultaneously for layered effects!

Parameter Naming Best Practices

‚úÖ Good Names ‚ùå Avoid Why
'speed' 's' Descriptive names are easier to identify
'particleCount' 'particle count' No spaces in parameter names
'colorHue' 'color-hue' Use camelCase, not dashes
'trailLength' 'length' Specific names prevent conflicts

üí° Code Examples

Example 1: Animated Circles

function setup() {
    createCanvas(windowWidth, windowHeight);
    colorMode(HSB, 360, 100, 100);
}

function draw() {
    // MIDI-controllable parameters
    let speed = p5Param('speed', 1.0, { min: 0.1, max: 5.0, label: 'Speed' });
    let size = p5Param('size', 100, { min: 20, max: 300, label: 'Size' });
    let hue = p5Param('hue', 180, { min: 0, max: 360, label: 'Color' });
    let trails = p5Param('trails', 0.1, { min: 0, max: 1, label: 'Trail Amount' });
    
    // Semi-transparent background for trails
    background(0, 0, 0, 1 - trails);
    
    // Animated position
    let x = width/2 + cos(frameCount * 0.02 * speed) * 150;
    let y = height/2 + sin(frameCount * 0.02 * speed) * 150;
    
    // Draw circle with MIDI-controlled properties
    fill(hue, 80, 100);
    noStroke();
    circle(x, y, size);
}

function windowResized() {
    resizeCanvas(windowWidth, windowHeight);
}

Example 2: Particle System

let particles = [];

function setup() {
    createCanvas(windowWidth, windowHeight);
    
    // Initialize particles
    for (let i = 0; i < 100; i++) {
        particles.push({
            x: random(width),
            y: random(height),
            vx: random(-2, 2),
            vy: random(-2, 2),
            life: random(100, 255)
        });
    }
}

function draw() {
    // Expose parameters
    let speed = p5Param('speed', 1.0, { min: 0.1, max: 3.0, label: 'Speed' });
    let gravity = p5Param('gravity', 0.1, { min: -0.5, max: 0.5, label: 'Gravity' });
    let particleSize = p5Param('size', 5, { min: 1, max: 20, label: 'Particle Size' });
    let fadeRate = p5Param('fade', 2, { min: 1, max: 10, label: 'Fade Rate' });
    
    background(0, 20);
    
    // Update and draw particles
    for (let p of particles) {
        // Apply gravity and speed
        p.vy += gravity;
        p.x += p.vx * speed;
        p.y += p.vy * speed;
        
        // Fade out
        p.life -= fadeRate;
        
        // Reset if dead or off-screen
        if (p.life <= 0 || p.x < 0 || p.x > width || p.y > height) {
            p.x = random(width);
            p.y = 0;
            p.vx = random(-2, 2);
            p.vy = random(-1, 1);
            p.life = 255;
        }
        
        // Draw particle
        fill(255, p.life);
        noStroke();
        circle(p.x, p.y, particleSize);
    }
}

‚úÖ Best Practices

Variable Declaration

  • Use let when storing parameter values
  • Avoid const - it prevents parameter updates
  • Call p5Param() in draw() for real-time updates

Parameter Ranges

  • Always specify min/max for predictable behavior
  • Use meaningful ranges - not just 0-1
  • Consider the parameter's purpose when setting limits

Performance

  • Limit parameter count - too many can clutter MIDI mappings
  • Use efficient code - P5 layer runs on top of 3D scene
  • Consider canvas size - full window can be performance-intensive

User Experience

  • Provide good labels - they appear in MIDI mapping UI
  • Use logical parameter names - easy to identify later
  • Test with MIDI controllers - ensure smooth parameter response

üîß Troubleshooting

‚ùå Parameters not updating with MIDI

  • Check variable declaration: Use let instead of const
  • Verify parameter names: Must match exactly in MIDI mapping
  • Check MIDI mapping: Ensure P5 dropdown has correct parameter selected
  • Look at console: Debug logs show parameter updates

‚ùå Parameters not appearing in MIDI dropdown

  • Run the sketch first: Parameters are registered when code runs
  • Check for syntax errors: Broken sketches don't register parameters
  • Refresh manually: Edit and re-run the sketch to refresh dropdowns

‚ùå Sketch not visible

  • Check layer visibility: Toggle in Layer Panel
  • Verify canvas size: Use windowWidth, windowHeight
  • Check z-index: P5 layer renders over 3D scene
  • Look for errors: Console shows compilation errors
